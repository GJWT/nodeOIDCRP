/**
 * State
 * Given state I need to be able to find valid access token and id_token
 * and to whom it was sent.
 * @class
 * @constructor
 */
class State {
  init(client_id, db, dbName, lifeTime) {
    db = db || null;
    dbName = dbName = '';
    lifeTime = lifeTime || 600;

    this.client_id = client_id;
    this.db = db;
    if (this.db == null) {
      if (dbName) {
        this.db = shelve.open(dbName, true);
      } else {
        this.db = {};
      }
    }
    this.lifeTime = lifeTime;
  }

  /**
   * Construct a state value. In this class it's just a random string.
   * Also store information about the request using the state value 
   * as key.
   * @param {string} receiver 
   * @param {Object<string, string>} request 
   * @param {State} state State value that someone else has created
   * @return A random string
   */
  createState(receiver, request) {
    let state = Math.random().toString(26).substring(2, 15);
    let now = Date.now();
    let stateInfo = {'client_id': this.client_id, 'as': receiver, 'iat': now};
    stateInfo = Object.assign(stateInfo, request);
    this.state = stateInfo;
    return state;
  }

  /**
   * Add information about an access token to the state information
   * @param {*} stateInfo The state information
   * @param {*} msg A response, typically a access token request response
   * @return The updated state information
   */
  updateTokenInfo(stateInfo, msg) {
    let tInfo = null;
    if (stateInfo['token']) {
      tInfo = stateInfo['token'];
    } else {
      tInfo = {};
    }

    let token = null;
    try {
      token = msg['access_token'];
    } catch (err) {
      console.log(err);
    }
    var exp = 0;
    if (token) {
      tInfo['access_token'] = token;
      if (msg['expires_in']) {
        exp = msg['expires_in'];
      } else {
        if (tInfo['expires_in']) {
          tInfo['exp'] = Date.now() + tInfo['expires_in'];
        }
      }

      if (exp) {
        tInfo['expires_in'] = exp;
        tInfo['exp'] = Date.now() + exp;
      }

      let claims = ['token_type', 'scope'];
      for (let i = 0; i < claims.length; i++) {
        let claim = claims[i];
        try {
          if (msg[claim]) {
            tInfo[claim] = msg[claim];
          }
        } catch (err) {
          console.log(err);
        }
      }

      stateInfo['token'] = tInfo;
    }
    return stateInfo;
  }

  /**
   * Add relevant information from a response to the state information
   * 
   * @param {string}response The response 
   * @param {State} state State value
   * @return state information
   */
  addResponse(response, state) {
    state = state || '';

    if (!state) {
      state = response['state'];
    }
    let stateInfo = this.state;
    if (!stateInfo) {
      stateInfo = this[state];
    }

    if (response['code']) {
      stateInfo['code'] = response['code'];
    }

    this.updateTokenInfo(stateInfo, response);
    let claims = ['id_token', 'refresh_token'];
    for (let i = 0; i < claims.length; i++) {
      let claim = claims[i];
      if (response[claim]) {
        stateInfo[claim] = response[claim];
      }
    }
    this.state = stateInfo;
    return stateInfo;
  }

  /**
   *  Add unspecific state information
   * 
   * @param {State} state State value
   * @return The present state information
   */
  addInfo(state, kwargs) {
    let stateInfo = this[state];
    stateInfo = Object.assign(stateInfo, kwargs);
    this[state] = stateInfo;
    return stateInfo;
  }

  /**
   * Bind a nonce value to a state value such that I later given a nonce
   * value I can find the state information
   * @param {string} nonce Nonce value
   * @param {State} state State value
   */
  bindNonceToState(nonce, state) {
    this.db['nonce_' + nonce] = state;
  }

  /**
   * Given a nonce value return the state value.
   * 
   * @param {string} nonce Nonce value
   * @return State value
   */
  nonceToState(nonce) {
    return this.db['nonce_' + nonce];
  }

  getTokenInfo(state, now, kwargs) {
    let tInfo = this[state]['token'];
    var exp = null;
    try {
      exp = tInfo['exp'];
    } catch (err) {
      console.log(err);
    }
    var now = null;
    if (!now) {
      now = Date.now();
    }

    if (now > exp) {
      console.log('Passed best before');
    }
    return tInfo;
  }

  /**
   * Get information about a access token bound to a specific state value
   * 
   * @param {State} state The state value
   * @param {*} now A timestamp used to verify if the token is expired or not
   * @return Token information
   */

  getTokenInfo(state, request, now, kwargs) {
    let tInfo = {};
    if (this.state && this.state['token']) {
      tInfo = this.state['token'];
    }
    try {
      var exp = tInfo['exp'];
    } catch (err) {
      console.log(err);
    }
    if (!now) {
      now = Date.now();
    }
    if (now > exp) {
      throw new Error('Passed best before');
    }
    return tInfo;
  }

  /**
   * Get the claims returned in a response connected to a specific state
   * value.
   * @param {State} state The state value
   * @param {ResourceRequest} request The type of response that is bound to a state
   * @param {date} now A time stamp
   * @return The response arguments
   */
  getResponseArgs(state, request, now, kwargs) {
    let sInfo = {};
    if (this.dict) {
      sInfo = this.dict[state];
    } else {
      sInfo = this[state];
    }
    request = new request();
    let tInfo = {};
    let reqArgs = {};
    now = now || 0;
    if (request) {
      for (let i = 0; i < Object.keys(request.cParam).length; i++) {
        let claim = Object.keys(request.cParam)[i];
        if (claim === 'access_token') {
          try {
            tInfo = this.getTokenInfo(state, now);
          } catch (err) {
            continue;
          }
          reqArgs[claim] = tInfo['access_token'];
        } else {
          if (sInfo[claim]) {
            reqArgs[claim] = sInfo[claim];
          }
        }
      }
    }
    return reqArgs;
  }

  getIdToken(state) {
    return this[state]['idToken'];
  }
}

module.exports.StateJWT = StateJWT;
